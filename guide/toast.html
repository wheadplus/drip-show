# toast 对话框

提供对话框交互

# 基础用法

点击查看代码 
// js
import plugin from "../../../src/components/toast/plugin";
import Vue from "vue";

// template
<d-button  @click="$toast('点击弹出提示',{autoClose:false})">上方弹出</d-button>
<d-button primary @click="$toast('点击弹出提示', {position:'middle'})">中间弹出</d-button>
<d-button error @click="$toast('点击弹出提示', {position:'bottom'})">下方弹出</d-button>

# 设置关闭按钮

点击查看代码 
// js
import plugin from "../../../src/components/toast/plugin";
import Vue from "vue";
methods: {
  onClickButton() {
    this.$toast("你知道我在等你吗？", {
      closeButton: {
        text: "知道了",
        callback: () => {
          console.log("他说知道了");
        }
      }
    });
  }
}

// template
<d-button  @click="onClickButton">上方弹出</d-button>

# 支持插入html标签

点击查看代码 
// js
import plugin from "../../../src/components/toast/plugin";
import Vue from "vue";
methods: {
  onClickButton() {
     onClickButton() {
      this.$toast('<strong style="color:red;">插入html标签</strong>', {
        enableHtml: true
      });
    }
  }
}

// template

# 属性

参数 说明 类型 可选值 默认值
autoClose toast关闭时间，默认5秒。可选择开关 Number,Boolean false/Number 5
position toast位置 String top/middle/bottom top
closeButton 关闭文字和关闭回调函数 Object
enableHtml 支持html标签，默认不开启 Boolean true false
